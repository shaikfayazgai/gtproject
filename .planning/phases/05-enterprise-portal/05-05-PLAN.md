     ```typescript
     import { EvidenceViewer } from '@glimmora/ui'
     import type { Evidence as ViewerEvidence } from '@glimmora/ui'
     import type { Evidence as PackEvidence } from '@glimmora/types'

     function toViewerEvidence(e: PackEvidence): ViewerEvidence {
       // Strip contributorId -- blind review boundary
       switch (e.type) {
         case 'code':
           return { type: 'code', language: 'typescript', code: e.content, filename: e.fileName }
         case 'file':
           return { type: 'document', filename: e.fileName || 'file', fileSize: String(e.fileSize || 'Unknown'), fileType: 'application/octet-stream', downloadUrl: e.fileUrl || '#' }
         case 'url':
           return { type: 'link', url: e.content, title: e.title, description: e.description }
         case 'video-url':
           return { type: 'video', url: e.content, title: e.title }
         case 'text':
           return { type: 'text', content: e.content }
       }
     }
     ```

**Step 2: Create BulkPaymentRelease using TanStack useReactTable directly.**

**`apps/enterprise-portal/src/components/projects/bulk-payment-release.tsx`:**
```
'use client'
```

IMPORTANT: @glimmora/ui DataTable encapsulates `rowSelection` state internally and does NOT expose it externally (no onSelectionChange or rowSelection props). Use TanStack `useReactTable` directly with external rowSelection state for this component.

```typescript
import { useReactTable, getCoreRowModel, getPaginationRowModel, flexRender, type ColumnDef } from '@tanstack/react-table'
import type { RowSelectionState } from '@tanstack/react-table'

const [rowSelection, setRowSelection] = useState<RowSelectionState>({})

const table = useReactTable({
  data: payments,
  columns: paymentColumns,
  state: { rowSelection },
  onRowSelectionChange: setRowSelection,
  getCoreRowModel: getCoreRowModel(),
  getPaginationRowModel: getPaginationRowModel(),
  enableRowSelection: (row) => row.original.status === 'pending',
})

const selectedPayments = table.getSelectedRowModel().rows.map(r => r.original)
const selectedCount = selectedPayments.length
const totalAmount = selectedPayments.reduce((sum, p) => sum + p.amount, 0)
```

Table columns: checkbox select, Milestone name, Amount (formatted currency), Status (Badge), Release Mode, Released Date (or "--").

Bulk payment release component:
1. Fetches GET /api/enterprise/projects/:id/payments via TanStack useQuery
2. Renders the custom table with useReactTable (row checkboxes only enabled for pending payments)
3. **Bulk action toolbar** rendered above the table when selectedCount > 0:
   ```html
   <div className="flex items-center justify-between p-3 bg-bg-card border border-border rounded-card mb-3">
     <span>{selectedCount} payment(s) selected</span>
     <Button variant="primary" onClick={handleBulkRelease}>
       Release Selected (${totalAmount})
     </Button>
   </div>
   ```
4. "Release Selected" button opens OTPConfirmationDialog with:
   - title: "Bulk Payment Release"
   - description: `Release ${selectedCount} payments totaling ${currency} ${totalAmount}? Enter your OTP to confirm.`
5. onConfirm calls POST /api/enterprise/payments/bulk-release with { paymentIds: selectedPayments.map(p => p.id), otp }
6. On success, clear selection (setRowSelection({})), invalidate payments query

7. Below the table, add sections:
   - **Auto-Payment Settings**: card with PaymentReleaseMode selector (manual / auto-on-approval / apg-silent) + apg-silent threshold amount input. Fetches GET and saves via PATCH /api/enterprise/payments/preferences
   - **APG-Silent Approvals Log**: DataTable showing automatically approved payments with amount, date, and threshold at time of approval. Fetches GET /api/enterprise/payments/silent-approvals

Props: `{ projectId: string }`
